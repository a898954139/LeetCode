public class Solution {
    public int RemoveDuplicates(int[] nums) {
        var d = new Dictionary<int,int>();
        foreach(var n in nums)
        {
            if(!d.ContainsKey(n))
                d[n]=0;
            d[n]++;
        }
        var a = d.Select(x=>x.Key).ToArray();
        Array.Sort(a);
        for(int i=0; i<a.Length; i++)
        {
            nums[i]=a[i];
        }
        return a.Length;
    }
}
